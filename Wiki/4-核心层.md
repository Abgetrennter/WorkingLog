### WorkLogItem（工作日志项）
**文件位置**: `WorkLogApp.Core/Models/WorkLogItem.cs:6-17`
```csharp
public class WorkLogItem
{
    public DateTime LogDate { get; set; }      // 事项日期
    public string ItemTitle { get; set; }      // 事项标题
    public string ItemContent { get; set; }    // 事项内容
    public int CategoryId { get; set; }        // 分类ID
    public StatusEnum Status { get; set; }     // 状态
    public DateTime? StartTime { get; set; }   // 开始时间
    public DateTime? EndTime { get; set; }     // 结束时间
    public string Tags { get; set; }           // 标签（逗号分隔）
    public int SortOrder { get; set; }         // 排序序号
}
```
